async function cargarProyectosDesdeJSON() { try { estado.innerText = "Procesando historial JSON... ⏳"; estado.style.color = "#f57c00"; const res = await fetch('history_all.json'); const rawHistoryData = await res.json(); if (rawHistoryData && rawHistoryData.length > 0) { processHistoryData(rawHistoryData); const datalist = document.getElementById('lista-proyectos'); datalist.innerHTML = ''; const mapaProyectos = new Map(); projectDatabase.forEach(p => { // 1. Ahora la clave incluye el año para evitar que se pisen entre ellos const claveProyecto = `${p.code}/${p.year}`; const textoFinal = `${claveProyecto} - ${p.client} - ${p.desc}`; if (mapaProyectos.has(claveProyecto)) { const textoExistente = mapaProyectos.get(claveProyecto); // Mantenemos el criterio de quedarnos con el texto más corto/limpio if (textoFinal.length < textoExistente.length) { mapaProyectos.set(claveProyecto, textoFinal); } } else { mapaProyectos.set(claveProyecto, textoFinal); } }); // 2. ORDENAMIENTO DE DOS NIVELES (Por año y luego por número) const listaOrdenada = Array.from(mapaProyectos.values()).sort((a, b) => { // Extraemos la parte antes del primer guion (ej. "014/25") const prefijoA = a.split('-')[0].trim(); const prefijoB = b.split('-')[0].trim(); // Separamos el número del proyecto y el año usando el slash (/) const [codigoA, anioA] = prefijoA.split('/'); const [codigoB, anioB] = prefijoB.split('/'); // Convertimos los textos a números reales matemáticos const numAnioA = parseInt(anioA, 10) || 0; const numAnioB = parseInt(anioB, 10) || 0; const numCodigoA = parseInt(codigoA, 10) || 0; const numCodigoB = parseInt(codigoB, 10) || 0; // Primero evaluamos el año. Si son distintos, ordenamos por año (24, luego 25...) if (numAnioA !== numAnioB) { return numAnioA - numAnioB; } // Si son del mismo año, entonces ordenamos por el número del proyecto return numCodigoA - numCodigoB; }); // Insertamos la lista definitiva en el HTML listaOrdenada.forEach(texto => { const option = document.createElement('option'); option.value = texto; datalist.appendChild(option); }); estado.innerText = "Base de datos cargada (JSON) 🟢"; estado.style.color = "#4CAF50"; } } catch (error) { console.error("Error al procesar el archivo history_all.json:", error); estado.innerText = "Error cargando archivo JSON 🔴"; estado.style.color = "#d32f2f"; } }
